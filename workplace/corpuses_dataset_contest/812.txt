#include <iostream>
#include <sstream>

int main()
{
  int num;

  std::istringstream("0123459") >> num;
  std::cout << num << std::endl; 

  std::istringstream("0123459") >> std::dec >> num;
  std::cout << num << std::endl; 

  std::istringstream("abcf123") >> std::hex >> num;
  std::cout << num << std::endl; 

  std::istringstream("7651") >> std::oct >> num;
  std::cout << num << std::endl; 

  

  return 0;
}
