#include <iostream>
#include <vector>

int main()
{
    
    int numberOfLines;
    std::cin >> numberOfLines;
    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n'); 
    
    
    std::vector<std::string> lines(numberOfLines);    
    for(int i = 0; i < numberOfLines; ++i)
    {
        std::getline(std::cin, lines[i]);
    }

    
    for(const auto& value : lines)
    {
        std::cout << value << "\n";
    }
}
